API Endpoints

// Get all students
GET /api/v1/students
// Headers
Authorization: "Bearer YOUR_API_KEY"

// Response
{
  "data": [
    {
      "id": 1,
      "name": "Emma Wilson",
      "grade": "10A",
      "gpa": 3.8
    }
  ],
  "total": 1250
}
// Get student by ID
GET /api/v1/students/123

// Response
{
  "id": 123,
  "name": "Emma Wilson",
  "email": "emma@school.com",
  "grade": "10A",
  "subjects": ["Math", "Science", "English"],
  "gpa": 3.8
}
// Create new student
POST /api/v1/students
Content-Type: "application/json"

// Request Body
{
  "name": "John Doe",
  "email": "john@school.com",
  "grade": "9B"
}
